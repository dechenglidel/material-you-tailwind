## Usage
```tsx
import { ButtonGroup } from '@/components/ui/button-group'
import { Button } from '@/components/ui/button'
```

```tsx
<ButtonGroup>
    <Button>Button 1</Button>
    <Button>Button 2</Button>
    <Button>Button 3</Button>
</ButtonGroup>
```
